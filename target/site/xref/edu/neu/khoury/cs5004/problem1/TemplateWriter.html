3   import java.io.File;
4   import java.io.FileOutputStream;
5   import java.io.IOException;
6   import java.io.OutputStreamWriter;
7   import java.io.PrintWriter;
8   import java.nio.charset.StandardCharsets;
9   
10  /**
11   * A class to handle all writing of Templates.
12   */
13  public class TemplateWriter implements ITemplateWriter {
14  
15    ParsedCsV parsedCsv;
16    ParsedTemplate parsedTemplate;
17    String outPutDirectory;
18    String templateType;
19  
20  
21    /**
22     * A constructor for a TemplateWriter object.
23     * @param parsedCsv parsedCsv
24     * @param parsedTemplate parsedTemplate
25     * @param outPutDirectory an output Dir
26     * @param templateType the type of template to write
27     */
28    public TemplateWriter(ParsedCsV parsedCsv,
29        ParsedTemplate parsedTemplate, String outPutDirectory, String templateType) {
30      this.parsedCsv = parsedCsv;
31      this.parsedTemplate = parsedTemplate;
32      this.outPutDirectory = outPutDirectory;
33      this.templateType = templateType;
34    }
35  
36    /**
37     * Populates a template with a list from an index.
38     *
39     * @param ind index
40     * @return populatedTemplate String
41     */
42    public String populateTemplate(Integer ind) {
43      String populatedTemplate = this.parsedTemplate.getTemplateString();
44      for (String field : this.parsedTemplate.getFields()) {
45  
46        populatedTemplate = populatedTemplate
47            .replaceAll("\\[\\[" + field + "]]", this.parsedCsv.getMap().get(field).get(ind));
48      }
49      return populatedTemplate;
50  
51    }
52  
53  
54    /**
55     * Writes autopopulated templates to a batch of files.
56     * @throws IOException IOException
57     */
58    public void writeToFile() throws IOException {
59      Integer row;
60      Integer number;
61      File file;
62      for (row = 0; row < this.parsedCsv.getNumRows(); row++) {
63        number = row + 1;
64        if (this.templateType.equals("--email-template")) {
65          file = new File(outPutDirectory + "emails" + number.toString());
66        } else {
67          file = new File(outPutDirectory + "letter" + number.toString());
68        }
69  
70        PrintWriter printWriter = new PrintWriter(new OutputStreamWriter(new FileOutputStream(file),
71            StandardCharsets.UTF_8));
72        printWriter.println(this.populateTemplate(row));
73        printWriter.close();
74      }
75  
76  
77    }
78  
79  
80  }
